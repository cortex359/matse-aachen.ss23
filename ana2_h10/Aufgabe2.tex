\section{Aufgabe 2}
Lösen Sie das folgende Anfangswertproblem
\begin{equation*}
    y'' - 3y' + 2y = \ct{e}^x \qquad y(0)=3, \quad y'(0)=4
\end{equation*}

\subsection{Lösung 2}

% Lösung durch WolframAlpha bestätigt
$$
    y(x) = \ct{e}^x \cdot \left(-x + 2 \ct{e}^x + 1\right)
$$
