\section{Aufgabe 5}
Bestimmen Sie die Lösung der folgenden Differentialgleichung
\[
    y' - 2y = (2\sin(x) + 5\cos(x)) \cdot \ct{e}^{-3x}
\]

\subsection{Lösung 5}
\begin{align*}
    y(x) &= c \cdot \ct{e}^{2x} - \frac{5}{26} \ct{e}^{-3 x} \sin(x) - \frac{27}{26} \ct{e}^{-3x} \cos(x)
\end{align*}
